Project Guide: The Blockchain Wish Jar

In this module, you will build a complete Decentralized Application from scratch. It's a "Wish Jar" where anyone in the world can store a wish on the blockchain forever by sending a small tip.

Learning Objectives

  • Write a complete, functional Smart Contract using Solidity.
  • Understand and use structs, arrays, events, and payable functions.
  • Compile and deploy your DApp to a live test network using the Remix IDE.
  • Interact with your deployed DApp by sending transactions.

Smart Contract: WishJar.sol

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract WishJar {
    address public owner;
    struct Wish {
        address person;
        string message;
        uint256 timestamp;
    }
    Wish[] public wishes;
    event NewWish(address indexed from, uint256 timestamp, string message);

    constructor() {
        owner = msg.sender;
    }

    function makeWish(string memory _message) public payable {
        require(msg.value >= 0.001 ether, "Send at least 0.001 ETH!");
        require(bytes(_message).length > 0, "Message cannot be empty!");
        wishes.push(Wish(msg.sender, _message, block.timestamp));
        emit NewWish(msg.sender, block.timestamp, _message);
    }

    function getAllWishes() public view returns (Wish[] memory) {
        return wishes;
    }

    function withdrawCoins() public {
        require(msg.sender == owner, "Only the owner can withdraw.");
        (bool success, ) = owner.call{value: address(this).balance}("");
        require(success, "Withdrawal failed.");
    }
}

Project Guide: Create & Mint Your First NFT Collection

This project takes you through the entire process of creating a real-world NFT collection. You will learn to handle artwork, upload to decentralized storage (IPFS), write an ERC-721 smart contract, and build a simple minting DApp.

Key Steps

  1. Prepare your artwork and upload it to a decentralized storage service like Pinata to get a CID.
  2. Create JSON metadata files for each art piece, pointing to the image's IPFS location.
  3. Upload the metadata folder to IPFS to get a second CID.
  4. Write and deploy the ERC-721 smart contract below using Remix.
  5. After deployment, call the `setBaseURI` function with your metadata folder's CID.
  6. Build a simple frontend with a "Mint" button to interact with your contract.

Smart Contract: CosmicAnimals.sol

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/Counters.sol";

contract CosmicAnimals is ERC721, Ownable {
    using Counters for Counters.Counter;
    Counters.Counter private _tokenIdCounter;
    uint256 public constant MAX_SUPPLY = 1000;
    uint256 public mintPrice = 0.01 ether;
    string public baseURI;

    constructor() ERC721("CosmicAnimals", "CANI") {}

    function safeMint() public payable {
        require(_tokenIdCounter.current() < MAX_SUPPLY, "All minted");
        require(msg.value >= mintPrice, "Not enough ETH sent");
        _tokenIdCounter.increment();
        uint256 tokenId = _tokenIdCounter.current();
        _safeMint(msg.sender, tokenId);
    }

    function _baseURI() internal view override returns (string memory) {
        return baseURI;
    }

    function setBaseURI(string memory _newBaseURI) public onlyOwner {
        baseURI = _newBaseURI;
    }

    function withdraw() public onlyOwner {
        (bool success, ) = owner().call{value: address(this).balance}("");
        require(success, "Withdrawal failed");
    }
}
